int KMP(char s[], char pattern[]){
	int i = 0, j = 0;
	int sLen = strlen(s);
	int patternLen = strlen(pattern);
	int * next = (int *)malloc(sizeof(int) * MaxSize);
	getNextVal(pattern, next);
	while(i < sLen && j < patternLen){
		/*匹配时一共可能出现3种状态*/
		if(s[i] == pattern[j]){	//状态1：匹配，同时往后挪
			++i;
			++j;
		}else if(j > 0){		//状态2：不匹配且前面有过已经匹配的，调整j
			j = next[j];
			if (j == -1) ++j;   //针对nextval中很多-1的情况，直接从0开始比
		}else{					//状态3：从第一个开始的地方不匹配，i往后挪
			++i;
		}
	}
	return (j == patternLen) ? i - j : -1;
}

建立next数组

  • 这里字符串都是从0开始存储的，最前面的next[0]设置成-1，如果是从1开始存储的那么next[1]设置成0，所有的next[]值比平常多1
  • next[]数组存的next[j]是最大公共前后缀中的最大公共前缀最后一个元素的后面元素的一个下标
  • 遇到匹配成功一部分之后突然失配，通过next[]数组直接调整到了下一个位置

void getNext(char pattern[], int next[]){
	int k, j;
	int patternLen = strlen(pattern);
	next[0] = -1;

	/*若pattern[next[j - 1] - 1 + 1] == pattern[j - 1]则next[j] = next[j - 1] + 1*/		
	/*k记录了next[j - 1] - 1 + 1*/
	for(j = 1, k = -1; j < patternLen; ++j){	

		/*一直往前找，直到到了最前面或是找到了前一种状态*/
		while(k >= 0 && pattern[k] != pattern[j - 1]) k = next[k];
		
		/*这之后分两种情况：k为-1(此时应该从0开始将next[j]设置为0)或找到的情况，发现两种可以合并起来*/
		next[j] = ++k;
	}
}

优化next数组为nextval数组

  • 在j通过next[j]往前跳的时候，如果调到前面的pattern上的字符和后面的一样，那就还需要继续往前跳，所以可以在这个上面做优化，如果pattern[next[j]]跟pattern[j]一样的话，next[j]直接取前面的next[next[j]]，这样就省去了中间那些同一字符多次重复的比较，减少了循环次数。

void getNextVal(char pattern[], int nextval[]){
	int k, j;
	int patternLen = strlen(pattern);
	nextval[0] = -1;
	for(j = 1, k = -1; j < patternLen; ++j){	
		while(k >= 0 && pattern[k] != pattern[j - 1]) k = nextval[k];	
		nextval[j] = ++k;
		
		/*j匹配失败时，当j这个位置与nextval[j]上的字符相同时，也肯定是匹配不了的，
		 所以就再往前找，去除无用的比较*/
		if (pattern[j] == pattern[nextval[j]]){
			nextval[j] = nextval[nextval[j]]; 
		}
	}
}
