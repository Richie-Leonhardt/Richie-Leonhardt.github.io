function fib(/*var*/ a){

//不带参数的sort方法是升序排字符串
var text = prompt("Input some numbers:","");
var order = prompt("order asc/desc","desc");
var word_list = text.split(/\s+/); //记住这个形式，字符类\s匹配所有空格字符[\r\t\n\f]

function ascending(a, b){return a - b;}   // < 0是需要调换位置，升序a - b,书本错了
function descending(a, b){return b - a;}

if (order === "asc")
	word_list.sort(ascending);	
else if (order === "desc")
	word_list.sort(descending);

alert(word_list.join(", "));	//这里直接省略或者换成toString()的话则默认是用逗号连接所有元素输出

// //数组的删除操作，老老实实使用传统迭代的方法最快，因为不是数据结构不需考虑时间复杂度
function no_zeros(arr){
	var	flag = false;
	for(var i = 0; i < arr.length; i++){
		if (arr[i] === 0) {
			/*for(var j = arr.length - 1; j > i ; j--) 倒过来移动作死啊*/
			for(var j = i + 1; j < arr.length; j++)
				arr[j - 1] = arr[j];
			arr.length--;
			i--;		//下一个的值覆盖了这个位置，需要回溯一下
			flag = true;
		}
	}
	/*if (flag) return true;
	  else return false; 傻屌写法！*/
	return flag;
}

function counter(arr){
	var count0 = 0,
		countNegative = 0,
		countPositive = 0;
	for (var i = 0; i < arr.length; ++i){
		switch(arr[i] / Math.abs(arr[i])){
			case 1:countPositive++;break;
			case -1:countNegative++;break;
			default:count0++;	//0作为被除数时为NaN，注意这里不能用case Number.NaN，判断NaN唯一方法是isNaN(n)
		}
	}
	alert("Positive Number:" + countPositive + '\n0:' + count0 + "\nNegative Number" + countNegative +"\n");
}

function row_averages(arr){
	var avr = new Array(arr.length);
	for (var i = 0; i < arr.length; ++i){
		var sum = 0;
		for(var j = 0; j < arr[i].length; ++j){
			sum += arr[i][j];
		}
		avr[i] = sum / arr[i].length;
	}
	return avr;
}

function reverser(n){
	var arr = new Array();
	while(n > 9){
		arr = arr.concat(n % 10);	//这边用push方法更好
		n /= 10;
		n = Math.floor(n);	//注意js中全部数值运算都是double类型的，做除法运算的时候将会有偏差，用Math.floor()消除，且也不会改变原值而是把结果返回来
	}
	arr = arr.concat(n);	//★★★注意：concat函数同C语言不一样，它不会改变原有数组，只会返回一个新的连接的数组
	alert(arr.join("\n"));
}
