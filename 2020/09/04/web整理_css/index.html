/*注意：CSS不能使用双斜杠//注释,无论单行多行只能像这样注释*/

/*通配选择器*/
 * {

}

/*简单选择器*/
h3{
	font-family:"Arial, Helvetica, Futura";
	font-size: 36px;
	font-style: italic; /*normal oblique(无用)*/
	text-align:center;
	height:auto;
}

/*类选择器,通过已有的标签创建不同的类*/
p.normal{
	font-variant: small-caps; /*normal 小型大写字母*/
}
p.warning{
	font-weight: bold; /*normal bolder lighter或100，200，……，900*/
}

/*通用选择器*/
.emphasize{
	/*最后一个是字体族，倒数第二是大小，前面俩随便*/
	font:bold italic 1.1em 'Times New Roman' Palatino;
	color: #FF0000;
}

.container{
	width: 800px;
	height: 600px;
	background-color: #FFFFFF;
	margin-left: 300px;
	padding: 50px;
}

/*id选择器*/
#ggg{
	text-decoration: line-through;  /*overline, underline 上划线中划线下划线*/

	letter-spacing: 1px; /*-1px nromal 字母间距(tracking)*/

	word-spacing: 5px; /*-1px normal 词间距*/

	line-height:2; /*24px 0.8em normal 行间距(leading)，数字代表倍数*/
}


/*上下文选择器：后代选择器，只要是后代就符合*/
ol ol{
	list-style-type: upper-alpha; /*=upper-latin lower-roman */
}
ol ol ol{
	list-style-type: decimal;
}


/*上下文选择器：子元素选择器，严格按照dom树结构*/
ul > li > li {
	list-style-type: square; /*circle空心圆 disc默认实心圆*/
}


/*选择第一个子元素*/
p:first-child{
	text-indent: 10px; /*2em 首行缩进*/
}

/*选择最后一个子元素*/
p:last-child{

}

/*选择子元素，且仅存在着这一个子元素*/
p:only-child{
	text-align: center; /*justify left(default) right 文本水平方向排列*/
}

/*选择自己：且自己没有子元素*/
p:empty{
	color: #777777;
	background-color: #FFFFFF;
}

/*伪类：link和visited，hover，focus*/


#ggg:hover{
	background-color: #FFFFFF;
}

a:link {color: #FF0000}     /* 未访问的链接 */
a:visited {color: #00FF00}  /* 已访问的链接 */
a:hover {color: #FF00FF}    /* 当有鼠标悬停在链接上 */
a:active {color: #0000FF}   /* 被选择的链接 */

/*图片悬浮*/
img {
	float: right;	/*图片悬浮位置，图片周围围绕文字的text-align要与float的相反才行*/
}

/*盒模型相关参数*/
td, th{
	border: thin solid black;  
	/*宽度(border-width) px
	  样式(border-style) solid dotted dashed double 
	  颜色(border-color)*/
}

table{
	border: thin solid black;
	border-collapse: collapse;
	border-top-width: medium;
	border-bottom-style: dashed; 
}

p.box{
	margin: 15px; 	/*外边距*/
	padding: 15px; 	/*内边距，注意是pad不是ped*/
	border-style: solid;
	background: #C0C0C0;
}


/*背景图片*/
body{
	background-image: url("../image/background.jpg");
	background-size: 1280px 720px; 
	background-position: top center;
	background-repeat: no-repeat; /*repeat(default) repeat-x repeat-y*/
}
