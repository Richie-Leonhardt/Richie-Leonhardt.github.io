<!DOCTYPE html>
<html lang="en">
<head>
    <title>A form simple</title>
    <meta charset="utf-8" />
</head>

<body>
    <!-- 页眉 -->
    <hgroup>
        <header>
            <h1>Practice2</h1>
            <h3>form、input、textarea、select、audio and video</h3>
        </header>
    </hgroup>
    
    <!-- 主要表单 -->
    <form action="" method="post">
        <!-- if no method, default get -->
        <!-- 两个输入框，输入用户名和密码 -->
        <p><!-- input tag need container -->
            <!-- pre使能够保留连续的多个空格和使用等距字体 -->
            <!-- text和password这两个，输入的值都是在value特性里面，可以用value特性事先指定好文本框内容 -->
            <pre>account: <input type="text" name="name" id="name" size="20" maxlength="25" /></pre>
            <br />
            <pre>password:<input type="password" name="pw" id="pw" size="20" maxlength="25" /></pre>
        </p>

        <!-- 多选checkbox -->
        <div>
            <p>
                choice(this is a checkbox)
            </p>
            <label> <input type="checkbox" name="bulb" id="bulb" value="25W * 4 $2.39" />25W * 4 $2.39
            </label>
            <label>
                <input type="checkbox" name="bulb" id="bulb" value="25W * 8 $4.29" checked="checked" />
                25W * 8 $4.29
            </label>
            <label>
                <input type="checkbox" name="bulb" id="bulb" value="good 25W * 4 $3.95" />good 25W * 4 $3.95
            </label>
            <label>
                <input type="checkbox" name="bulb" id="bulb" value="good 25W * 8 $7.49" checked="checked" />good 25W * 8 $7.49
            </label>
        </div>

        <!-- 单选 -->
        <p>payment(this is a radio)</p>
        <label><input type="radio" name="pay" id="pay" value="Visa" checked="checked">Visa</label>
        <label><input type="radio" name="pay" id="pay" value="MasterCard">MasterCard</label>
        <label><input type="radio" name="pay" id="pay" value="Discover">Discover</label>

        <!-- 单选菜单形式 -->
        <p>
            select one:
            <select name="choice1">
                <option>ggg</option>
                <option>hhh</option>
                <option>xxx</option>
            </select>
        </p>

        <!-- 多选菜单形式 -->
        <p>
            select one or more:
            <select name="choice2" multiple="multiple">
                <!-- multiple choices -->
                <option>ggg</option>
                <option>hhh</option>
                <option>xxx</option>
            </select>
        </p>

        <!-- textarea输入文本 -->
        <!-- 如果要在一段文字中限制每行的字符，使用textarea是一个很好的方式 -->
        <p>
            <textarea name="ta1" rows="3" cols="20">default txt here</textarea>
        </p>

        <!-- 提交和重置按钮 -->
        <p>
            <input type="submit" name="sub" value="Submit Form" />
            <input type="reset" name="reset" value="Reset Form" />
        </p>
    </form>

    <!-- 添加音乐 -->
    <audio controls="controls">
        <!-- no autoplay -->
        <source src="audio/三亩地 - 寻.mp3" />
        <source src="audio/三亩地 - 寻.wav" />
        <source src="audio/三亩地 - 寻.ogg" />
        don't support this audio element
    </audio>

    <!-- 添加视频 -->
    <video width="1280" height="760" controls="controls" preload="preload" autoplay="autoplay">
        <source src="video/test.mp4" />
        <source src="video/test.ogv" />
        <source src="video/test.webm" />
    </video>

    <!-- 页脚 -->
    <footer>
        &copy; Ritchie Leonhardt
        <br />
        2020
    </footer>
</body>
</html>
