P2 JS

function colorChoice(){
	var color;
	var dom = document.getElementById("myForm");
	for (var i = dom.length - 1; i >= 0; i--) {
		if(dom.colorButton[i].checked){
			color = dom.colorButton[i].value;
		}
	}

	switch(color){
		case "red":alert("red");break;
		case "orange":alert("orange");break;
		case "green":alert("green");break;
		case "blue":alert("blue");break;
		default:alert("error");
	}
}

var dom = document.getElementById("myForm");
dom.elements[0].onclick = colorChoice;
dom.elements[1].onclick = colorChoice;
dom.elements[2].onclick = colorChoice;
dom.elements[3].onclick = colorChoice;

注意多选的checkbox实际上是一个数组，数组名即为标签的name
function countPrice(){
	var dom = document.getElementById("myForm");
	var sum = 0;
	for (var i = 0; i < dom.fruitsCheck.length; i++) {	//这边是dom.fruitsCheck.length而不是dom.fruitsCheck.elements.length	
		if(dom.fruitsCheck[i].checked){
			sum += Number(dom.fruitsCheck[i].value);	//默认+都是转换成字符串的，所以需要转成Number	
		}
	}
	alert('Your total cost is:' + sum);
	return false;
}

document.getElementById("submitButton").onclick = countPrice;	//注意！这里是不同加括号直接函数名注册的

P4_5_6 JS

function countPrice(){
	var dom = document.getElementById("myForm");
	var sum = 0;
	var price = [0.59, 0.49, 0.39];
	var errorTag = false;
	//多个name都相同且处于同一级别，构成了一个数组length为3的fruitsNumber[]数组，可以这样使用
	for (var i = 0; i < dom.fruitsNumber.length; i++) {
		var str = dom.fruitsNumber[i].value;
		if(str != ""){
			//if(str.search(/^[0-9]{1,2}$/) != -1){		//注意，正则表达式这里是不加引号的，否则岂不是成字符串了
			if (Number(str) >= Number(dom.fruitsNumber[i].getAttribute("min")) && Number(str) <= Number(dom.fruitsNumber[i].getAttribute("max"))){
				sum += dom.fruitsNumber[i].value * price[i];		
			}else{
				dom.fruitsNumber[i].value = "";
				dom.fruitsNumber[i].focus();	//书中并不推荐，并不是所有浏览器普遍适用
				errorTag = true;
			}
			//}
		}
	}
	if (errorTag) {
		alert("input error, please input number between " + dom.fruitsNumber[0].getAttribute("min") + "-" + dom.fruitsNumber[0].getAttribute("max"));
		return false;
	}
	alert('Your total cost is:' + sum);
	return false;
}
document.getElementById("submitButton").onclick = countPrice;

P7 JS

function checkName(){
	if(this.value != "" && this.value.search(/^[A-Z][a-z]* +[A-Z][a-z]* +[A-Z][a-z]*$/) == -1){
		alert("incorrect name, please input again");
		this.value = "";
		this.focus(); //不推荐用
	}
}

function checkAge(){
	if(this.value != "" && Number(this.value) < 18){
		alert("age should greater than 17");
		this.value = "";
		this.focus(); //不推荐用
	}
}

function checkWeight(){
	if(this.value != "" && Number(this.value) < 80 || Number(this.value) > 300){
		alert("incorrect Weight, please input again");
		this.value = "";
		this.focus(); //不推荐用
	}
}

document.getElementById("userName").onblur = checkName;
document.getElementById("userAge").onblur = checkAge;
document.getElementById("userWeight").onblur = checkWeight;

function colorChoice(event){
	var color  = event.currentTarget;
	switch(color.value){
		case "red":alert("red");break;
		case "orange":alert("orange");break;
		case "green":alert("green");break;
		case "blue":alert("blue");break;
		default:alert("error");
	}
}
//因为colorChoice是一个数组，所以在加监听器的时候也使用数组的形式
var choice = document.getElementById("myForm").colorChoice;
for (var i = 0; i < choice.length; i++) {
	choice[i].addEventListener("click", colorChoice, false);
}
