//要求原来的顶层往下移动时，应该回到原来的位置
//这样的话应该是顶层设成15 然后10 5 0为原来位置，对号入座就行

var ttop = 0;

//这里修改成代号的形式,方便zIndex的对上
function showTop(x){
	var oTop = document.getElementById('t' + ttop).style;
	var toTop = document.getElementById('t' + x).style;
	toTop.zIndex = '15';
	oTop.zIndex = String(5 * ttop);	//对应关系
	ttop = x;
}

//p7 其他题意理解的
//要求原来的顶层往下移动时，下移一层
//以数组下标为zIndex层次
//通过数据结构基础的分析，每次抽出来那个到顶的，然后从这个下标开始后面的元素往前移动一位
//最后将这个目标元素插到最后一位就行了，然后遍历所有元素将下标值赋给元素的style.zIndex

var idArr = ['t0','t1','t2'];
function showTop(x){
	for(var i = 0; i < idArr.length; ++i){
		if(idArr[i] == 't' + x){
			for(var j = i; j < idArr.length - 1; ++j){
				idArr[j] = idArr[j + 1];
			}
			idArr[idArr.length - 1] = 't' + x;
			break; 
		}
	}
	for(var i = 0; i < idArr.length; ++i){
		var dom = document.getElementById(idArr[i]).style;
		dom.zIndex = String(i);
	}
}

//一个很重要的点：style在HTML中没有设置的话，那个属性在JavaScript中不能操作，赋值也没用
//所以一定要牢记：在使用JavaScript操作style的值的时候，一定确保这些已经在HTML或者CSS中初始化过
function mover(){
	var x = parseInt(document.getElementById("x").value);
	var y = parseInt(document.getElementById("y").value);
	move1px(x, y);
}

function move1px(x, y){
	var dom = document.getElementById("image1").style;
	var cx = parseInt(dom.left);
	var cy = parseInt(dom.top);
	if (cx != x || cy != y) {
		if (cx < x) cx++;
		else cx--;
		if (cy < y) cy++;
		else cy--;

		dom.top = cy + "px";
		dom.left = cx + "px";
		setTimeout(move1px, 10, x, y);
	}
}
// move再改：以上每次移动一个单位的方法只能往固定的八个方向移动
// 做出改进，使通过时间和坐标的函数来达到任意方向直线移动的效果
// 
//在web应试中，但凡遇到多个函数共用参数，都使用全局变量
//这边因为操作的对象只有一个dom，所以设置成全局变量能够改善性能
var k, times, x, y, ox, oy, ct, dom, t;

function mover(){
	x = parseInt(document.getElementById("x").value);
	y = parseInt(document.getElementById("y").value);
	ox = parseInt(document.getElementById("image1").style.left);
	oy = parseInt(document.getElementById("image1").style.top);
	k = (y - oy)/(x - ox);
	times = 100;	//这边设置的是经过一百步之后到达终点
	ct = 0;
	// moveTimer(); 
	dom = document.getElementById("image1").style;
	t = setInterval(moveTimer1, 10);
}

function moveTimer(){
	if (ct < times) {
		dom.top = Math.ceil((ct / times) * (y - oy) + oy) + 'px';
		dom.left = Math.ceil((ct / times) * (x - ox) + ox) + 'px';	//切记，floor和ceil函数不能省略Math
		ct++;
		setTimeout(moveTimer,10,ct);	//如果步数太多占用的内存就多了，所以在性能上还是推荐setInterval
	}
}

//换一种实现方式，使用setInterval而不是setTimeout递归
//这边内部clearInterval是没有问题的，clearInterval书上没讲，算是拓展了
function moveTimer1(){
	if (ct > times) clearInterval(t); //这个函数要求延迟事件的这个对象作为参数
	dom.top = Math.ceil((ct / times) * (y - oy) + oy) + 'px';
	dom.left = Math.ceil((ct / times) * (x - ox) + ox) + 'px';	//切记，floor和ceil函数不能省略Math
	ct++;
}

var x, y;	//x,y 是鼠标起始点与图片原点（即左上角位置的相对位置）

/*在外面这样使用会出错：Uncaught TypeError: Cannot read property 'style' of null*/
//var dom = document.getElementById("image1").style;

// 正确的应该是这样,然后在grabber函数里面定位元素
var dom;

function grabber(event){
	dom = event.currentTarget.style;
	x = parseInt(dom.left) - parseInt(event.clientX);
	y = parseInt(dom.top) - parseInt(event.clientY);
	document.addEventListener('mouseup', dropper, true); //就一个元素，true和false都行
	document.addEventListener('mousemove', mover, true);
}

function mover(event){
	dom.left = (parseInt(event.clientX) + x) +'px';
	dom.top = (parseInt(event.clientY) + y) + 'px';
	event.stopPropagation();	//用的是mousemove，这边的阻止传播是必要的！
	event.preventDefault();
}

function dropper(event){
	event.stopPropagation();	//非必要但以防万一还是加上好一点
	event.preventDefault();
	document.removeEventListener('mousemove', mover, true); //松开鼠标保证一切恢复原样
	document.removeEventListener('mouseup', dropper, true); 
}
