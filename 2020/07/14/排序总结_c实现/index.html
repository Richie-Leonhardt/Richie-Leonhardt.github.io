快速排序

//快速排序
#include <stdio.h>
#define N 8

//左右搜索，将比枢轴大的元素移到后面，比枢轴小的移到前面，可以通过交替找的方法将那个枢轴空出来的位置充分利用起来，不交换而采取移动的方式
int Partition(int sqList[], int low, int high){
	int pivokey = sqList[low];
	while( low < high ){
		while ( low < high && sqList[high] >= pivokey ) high--;	//因为一直往前移动搜索比枢轴小的数字，如果搜到头都没找到，则会出现越界，所以一定要加上low < high的条件不能少
		sqList[low] = sqList[high];
		while ( low < high && sqList[low] <= pivokey ) low++;	
		sqList[high] = sqList[low];
	}
	/*最后枢轴返回位置的时候要考虑到，没执行while和执行了while的情况，执行了while就要找中间那个位置，发现high和low都行，但是若没执行while就必须low了，所以最后必须是low*/
	sqList[low] = pivokey;
	return low;
}

//每次分割开来左右递归
void Qsort(int sqList[], int low, int high){
	int pivoloc;
	if(low < high){
		pivoloc = Partition(sqList,low,high);
		Qsort(sqList, low, pivoloc - 1);
		Qsort(sqList, pivoloc + 1,high);
	}
}

void printList(int sqList[]){
	int i;
	for ( i = 0; i < N; ++i )
		printf("%d ",sqList[i] );
}

int main(int argc, char const *argv[])
{
	int i;
	int sqList[N];
	printf("请输入8个数字，以空格隔开：");
	for (i = 0; i < N; i++)
		scanf("%d",&sqList[i]);
	Qsort(sqList,0,N-1);
	printList(sqList);
	return 0;
}

基数排序

/* 基数排序
 * 基数排序是针对非负整数或是非正整数的序列进行的排序
 * 基数排序是先根据个位数进行排序，再根据十位数进行排序，再百位数……
 * 基数排序的基本过程：
 * 1.先找到最大的数有几位
 * 2.根据个位数分到不同的桶子里去，按序倒出来，再跟据十位数分到不同的桶子里去，按序倒出来，再百位数……
 * 3.最后一个分完之后从桶子里倒出来的时候，排序就完成了
 * 这边每次扫描要算桶子的数量，用数组来表示桶子的话就是二维数组，会导致很大一部分空间浪费（如果所有元素都集中到一个桶子的话）
 * 所以实现该算法的一个很好的方法是使用链表
 * 所以，整体读入数据到一个数组中，为了使之成为一个稳定的排序，且建立链表的时候采用头插法，所以先将读入的所有数据放到一个数组中，然后从数组尾部到头部头插法建立链表再执行排序
 */

#include <stdio.h>
#include <stdlib.h>
#include <math.h>

typedef enum {true,false} bool;
typedef struct node *node_pointer;
struct node{
    int data;
    node_pointer next;
};

node_pointer listArr[10];   //from 0 to 9

void insertAtTail(node_pointer list, node_pointer p){
    node_pointer q = list;
    while(q->next != NULL) q = q->next;
    q->next = p;
}

/*drop into the specific bucket found through num % 10^(digit)*/
void distribute(node_pointer masterlist, int digit){
    for (int i = 0; i <= 9; ++i){
        listArr[i]->next = NULL;
    }
    node_pointer p = masterlist->next;
    node_pointer q;
    while(p != NULL){
        q = p;
        p = p->next;
        insertAtTail(listArr[(int)(floor(q->data / pow(10, digit))) % 10], q);
        q->next = NULL;
    }
}

void coalesce(node_pointer masterlist){
    masterlist->next = NULL;    //clear the masterlist isn't necessary
    node_pointer p = masterlist;
    for (int i = 0; i <= 9 ; ++i){
        p->next = listArr[i]->next;
        while(p->next != NULL) p = p->next;
    }
}

void radixSort(node_pointer masterlist, int numdigits){
    int i;
    for (i = 0; i < numdigits; ++i){
        distribute(masterlist, i);  //drop into bucket
        coalesce(masterlist);   //drop out from bucket
    }
}

void insertAtFirst(node_pointer list, int X){  //if there is a list with head, then we could not return because it operate the nodes after head node
    node_pointer p = NULL;  //even use list doesn't matter, I should keep a good habit
    p = malloc(sizeof(struct node));
    p->data = X;
    p->next = list->next;
    list->next = p;
}

node_pointer initList(int arr[], int Capacity){
    node_pointer head = malloc(sizeof(struct node));//make a head node
    head->next = NULL;
    for (int i = Capacity - 1; i >= 0; --i){    //here easily make a bug, for the first element is the capcity - 1
        insertAtFirst(head, arr[i]);
    }
    return head;
}

void printList(node_pointer list){
    node_pointer p = list->next;
    while( p != NULL && p->next != NULL ){
        printf("%d ", p->data);
        p = p->next;
    }
    if (p != NULL) printf("%d", p->data); //there won't a space at the end of line
    printf("\n"); //if it's an empty list, an empty line could represent it
}

/* 堆排序
 * 堆排序是利用小顶堆大顶堆来进行排序，学好堆就很容易排序了
 * 这里使用小顶堆进行排序，首先是创建并初始化堆，然后插入元素的方法建好一个堆，最后边删除边调整堆
 * 另外，堆是不稳定的排序，末尾的元素到第一个元素后会打乱相对次序，举个反例就知道了
 */

#include<stdio.h>
#include <stdlib.h>
#define MINIUM -10001 //哨兵元素

typedef enum {true,false} bool;
typedef int ElementType;
typedef struct HeapStuct *MinHeap; 
struct HeapStuct{
    ElementType *Elements;  //存储堆元素的数组的头指针
    int Size;               //堆的当前元素个数
    int Capacity;           //堆的最大容量
};

MinHeap Create( int MaxSize ){
    MinHeap H = malloc(sizeof(struct HeapStuct));
    H->Elements = malloc(( 2 * MaxSize + 1 ) * sizeof(ElementType));
    H->Size = 0;
    H->Capacity = MaxSize;
    H->Elements[0] = MINIUM;
    return H;
}

bool IsFull( MinHeap H ){
    if ( H->Size == H->Capacity)
        return true;
    return false;
} 

bool IsEmpty( MinHeap H ){
    if (H->Size == 0)
        return true;
    return false;
}

//T(N) = O(logN)
void Insert( MinHeap H, ElementType item ){
    int i;
    if (IsFull(H) == true){
        //printf("MinHeap is Full!\n");
        return;
    }
    i = ++H->Size;
    /*由于这边存在ELements[0]为哨兵元素是最大值的上界，所以不用担心i一直往前与Elements[0]交换的情况,否则终止条件中应加i != 1*/
    for ( ; H->Elements[i/2] > item; i /= 2 ){  //经检验，不管末尾是左子树还是右子树，/2都没毛病
        H->Elements[i] = H->Elements[i/2];      //将小于item的往子树移动一位
    }
    H->Elements[i] = item;
}

//堆中的删除相当于队列中的出队列，堆中是树根的最大值删除并返回，下面的子元素补上来
//子元素补上来是要左右比较上来，再调整子树吗？同时还要保证调整后的最后一个那个位置空出来才能算完全二叉树
//肯定不行，要补上来，倒不如把最后一个补上来，然后跟插入一样，往下交换的移动方式弄好，注意往下交换，要保证交换的元素大于左右子树，不能像插入那样单边
ElementType DeleteMin( MinHeap H ){
    int X,tmp,i = 1;
    if ( IsEmpty(H) == true ){
        //printf("MinHeap is Empty!\n");
        return -10001;
    }
    X = H->Elements[1];
    tmp = H->Elements[H->Size--];

    /*存在左右子树的时候且左右子树有能够上位的，这个tmp要一直下沉，直到左右子树都没有或只有一个左子树的时候*/
    while ( i*2+1 <= H->Size && (H->Elements[i*2] < tmp || H->Elements[i*2+1] < tmp) ){
        if ( H->Elements[i*2] < H->Elements[i*2+1]){
            H->Elements[i] = H->Elements[i*2];
            i *= 2;
        }else{
            H->Elements[i] = H->Elements[i*2+1];
            i = i * 2 + 1;
        }
    }
    
    //最后没有右子树的情况单独拿出来，如果符合条件左子树与根交换位置，因为是完全二叉树不会有下一层，所以这里就是单独的循环
    if (i*2 == H->Size && H->Elements[i*2] < tmp){
        H->Elements[i] = H->Elements[i*2];
        H->Elements[i*2] = tmp;  
    }else{
        H->Elements[i] = tmp;
    }
    return X;
}

//表排序实现，用到静态链表
#include <stdio.h>
#include <stdlib.h>
#define MAXINT 10001
typedef enum {ture,false} bool;
typedef int ElementType;
typedef struct Node *SList;
typedef struct Node{
    ElementType Key;   //数据域
    int Next;         //指向父节点的下标，对于根节点设置成负数，负多少意味着它底下有多少节点
}SListType;

void LinkListSort(SListType S[],int length){
    int i, p, tmp;
    S[0].Next = 1;      //S[0].Key是最大值，S[0].Next 一直存储指向最小的指针
    S[1].Next = 0;
    for ( i = 2; i <= length; ++i){
        p = 0;
        if (S[i].Key < S[S[0].Next].Key){    //插入排好序的循环链表可分为三种情况：比最开始的小，比最后的大 中间的情况
            S[i].Next = S[0].Next;
            S[0].Next = i;
            continue;
        }

        /*下面的while中加了这个等号使表排序成为稳定的排序*/
        while (S[i].Key >= S[S[p].Next].Key){    //由于安排好最大的哨兵节点，所以后两种情况合一，一直往后搜索
            p = S[p].Next;
        }
        S[i].Next = S[p].Next;
        S[p].Next = i; 
    }
    p = S[0].Next;
    for (i = 1; i <= length; ++i){
        while(p < i)                //p在i前面的话说明是已经跟前面调换位置过的，这时候就要沿着指针再往后找直到p >= i找到
            p = S[p].Next; 

        tmp = S[i].Key;
        S[i].Key = S[p].Key;    
        S[p].Key = tmp;
        
        tmp = S[p].Next;
        S[p].Next = S[i].Next;
        S[i].Next = p;
        p = tmp;
    }
}

void printSList(SListType S[],int length){
    int i;
    for (i = 1; i <= length; ++i)
        printf("%d ", S[i].Key);
    printf("\n");
}

int main(int argc, char const *argv[]){
    int N,i;
    ElementType X;
    scanf("%d\n",&N);     //读入要排序的数目 题目要求8个整数
    SList S = malloc(( N + 1 )*sizeof(struct Node));
    S[0].Key = MAXINT;
    for(i = 1; i <= N;i++){
        scanf("%d",&X);
        S[i].Key = X;
    }
    LinkListSort(S,N);
    printf("The sorted static link list is : ");
    printSList(S,N);
    return 0;
}

归并排序

/* 归并排序
 * 归并排序是一种采用分而治之思想的排序
 * 基本步骤为：
 * 1.将数组划分为两个各包含n/2个项目的子数组
 * 2.除非数组比较小，否则再分为两份递归方式完成任务
 * 3.最后到最小单位之后，将子数组合并为单个有序的数组
 */

#include <stdio.h>
#include <string.h>
#define MAX_SIZE 100

void concatArr(int arr[], int leftArr[], int rightArr[], int leftLen, int rightLen){
    int i = 0, j = 0, k = 0;
    while(i < leftLen && j < rightLen){
        if (leftArr[i] <= rightArr[j]) arr[k++] = leftArr[i++];
        else arr[k++] = rightArr[j++];
    }
    while (i < leftLen) arr[k++] = leftArr[i++];
    while (j < rightLen) arr[k++] = rightArr[j++];
}

void mergeSort(int arr[], int size){
    if (size > 1){
        const int mid = size/2;
        const int midplus = size - mid;
        int leftArr[mid];
        int rightArr[midplus];
        memcpy(leftArr, arr, mid*sizeof(int));  //memcpy是string.h中的函数，复制第二个指针开始第三个参数个字节到第一个指针的内存上覆盖原来的
        memcpy(rightArr, arr + mid, midplus*sizeof(int));
        mergeSort(leftArr, mid);        //左递归继续分
        mergeSort(rightArr, midplus);   //右递归继续分
        concatArr(arr, leftArr, rightArr, mid, midplus);  //归并，连接两个有序数组
    }
}

void printArr(int arr[], int len){
    for (int i = 0; i < len-1; ++i) printf("%d ", arr[i]);
    printf("%d\n", arr[len - 1]);
}

int main(int argc, char const *argv[]){
    int size,i = 0;
    int arr[MAX_SIZE];
    scanf("%d\n", &size);
    while(scanf("%d", arr + i++) != EOF && getchar() != '\n');
    mergeSort(arr, size);
    printArr(arr, size);
    return 0;
}

/*实际上就是在原本插入排序的基础上加入了二分查找*/

int binarySearch(int key, int arr[], int len){
    int index;
    int left = 0;
    int right = len - 1;
    int mid = (left + right) / 2 ;
    while(left <= right && arr[mid] != key){
        if (arr[mid] > key){
            right = mid  - 1;
        }
        else {
            left = mid + 1;
        }
        mid = (left + right) / 2;
    }
    index = mid;
    if (arr[mid] != key){
        index = -1;
    }
    return index;//仍坚持单一出口的原则
}

void binaryInsertSort(int arr[],int len){
    for (int i = 1; i < len; i++){
        int temp = arr[i];
        int left = 0;
        int right = i - 1;
            while (left <= right){//这个等于，非常重要当相等时left和right和mid重合，这时候要判断left移动还是right移动
                int mid = (left + right) / 2;

                /*下边这个 = 一定要有，是在相等的时候排到后面一位，确保排序的稳定性
                最后的三个下标重合的时候，如果temp大则从mid的后一位往后开始移，如果temp小的话则就从mid往后开始移
                根据循环发现恰恰left能满足后移的条件，temp大的话left会后移，temp小left就不动，所以后移的条件是>= temp
                */
                if (temp >= arr[mid]){
                    left = mid + 1;//只有这边的 + 1和下面的 - 1能够保证最后三个下标能够重合
                }else {
                    right = mid - 1;
                }
            }
        for (int j = i - 1;j >= left;j--){
                arr[j + 1] = arr[j];
        }
        arr[left] = temp;
    }
}

插入排序与希尔排序

/**插入排序
 * 是以有序数列为基础的，将未排序的元素插入到有序数列的相应位置
 * 从i后面的一个元素开始一个一个往前比较直到0，这个元素位置就正确了
 * 插入排序是稳定的排序，并没有改变相对的位置
 * */
void insertSort1(int array[],int length){
    int temp;
    for (int i = 1;i < length;i++){
        temp = array[i];
        for (int j = i - 1;j >= 0;j--){
            if (j == 0 && array[j] > temp){ //为插入到下标为0的位置单独设一个
                array[1] = array[0];
                array[0] = temp;
                break;
            }
            if (array[j] > temp){ 
                array[j + 1] = array[j];
            }
            else {
                array[j + 1] = temp; //即使相等时也插入到了原来的后面，没有改变相对位置
                break;
            }
        }
    }
}

/*简化以上代码*/
void insertSort2(int array[],int length) {
    int temp;
    for (int i = 1; i < length; i++) {
        temp = array[i];
        int j;
        for (j = i - 1; j >= 0 && array[j] > temp; j--) {
            array[j + 1] = array[j];
        }
        //此时因为上面的判断所以应该补上少的那一位，完美地出现了j = -1的情况，修补了插入在初始位置的漏洞
        array[j + 1] = temp;
    }
}

/**希尔排序：间隔d的元素分成一组，组之间用插入排序，然后d/2不断重复直到d == 1，这时肯定已经排好了
 * 算是不断趋向于线性
 * 希尔排序是不稳定的排序，虽然插入排序稳定，但是再d变小的时候的再重整会打乱相对次序
 * */
void shellSort1(int array[],int length){
    int d = length;
    while ( d != 1 ){
        d = d / 2;
        for (int x = 0;x < d;x++) {//x的意思其实是组别，每一个d中都有x个组
            for (int i = x; i < length - d; i += d) {
                int temp = array[i + d];
                for (int j = i; j >= 0; j -= d) {
                    if (j == x && array[j] > temp){//这边的j = x是从上面的循环中推导出来的，原来的插入排序是j == 0的条件，现在是分组了
                        array[j + d] = array[j];
                        array[j] = temp;
                        break;
                    }
                    if (array[j] > temp) {
                        array[j + d] = array[j];
                    } else {
                        array[j + d] = temp;
                        break;
                    }
                }
            }
        }
    }
}

/**根据insertSort2依样画葫芦*/
void shellSort2(int array[],int length){
    int d = length;
    while( d != 1 ){
        d = d / 2;
        for (int x = 0;x < d;x++){
            for (int i = x + d;i < length;i += d){
                int temp = array[i];
                int j;
                for (j = i - d;j >= 0 && array[j] > temp;j -= d){
                    array[j + d] = array[j];
                }
                array[j + d] = temp;
            }
        }
    }
}

/*计数排序*/
#include <stdio.h>
#define MAX_SIZE 1000

void countSort(int arr[], int length){
    int max = -99999, min = 99999;
    for ( int i = 0; i < length; ++i) {
        if (arr[i] > max) max = arr[i];
        if (arr[i] < min) min = arr[i];
    }
    const int size = max - min + 1;
    int deck[size];
    for (int i = 0; i < size; ++i) deck[i] = 0;
    for (int i = 0; i < length; ++i) deck[arr[i] - min]++;
    for (int i = 0, j = 0; i < size; ++i){
        while(deck[i] != 0){
            arr[j] = i + min;
            j++;
            deck[i]--;
        }
    }
}

void printArray(int arr[], int length){
    for (int i = 0; i < length - 1; ++i) printf("%d ", arr[i]);
    printf("%d", arr[length - 1]);
}

int main(int argc, char const *argv[])
{
    int arr[MAX_SIZE];
    int tmp = 0;
    while(scanf("%d", &arr[tmp]) != EOF && getchar() != '\n') tmp++;
    countSort(arr, tmp + 1);
    printArray(arr, tmp + 1);
    return 0;
}

桶排序

/* 桶排序
 * 桶排序和计数排序有点像，如果说计数排序分成一个个“点”来通过点计数的话，
 * 那么桶排序就是分成一个个“段”，然后段内再递归调用自己或是使用另一种排序方法
 * 在排序的元素都为整数的情况下，计数排序和桶排序都是行之有效的，
 * 但是如果排序的元素中存在浮点数，计数排序就吃瘪了^_^
 * 所以，这里的代码针对浮点数数组的排序
 * 桶排序适用的情况是元素几乎平均分布的情况，就是说如果每一个桶装一个元素的话，时间复杂度能够到达O(N)
 * 这里，每隔1分一个桶，也就是说分为floor(max) - floor(min) + 1 个区间
 * 第一个桶的区间就是[floor(min), floor(min + 1)),最后一个桶的区间就是 [floor(max), floor(max + 1))
 * 桶内的元素使用链表实现，然后桶内部的排序采用就采用插入排序，比较适合链表的特性
 * 这边是升序排序且稳定的排序
 * */
#include <stdio.h>
#include <math.h>
#include <stdlib.h>
#define MAX_SIZE 100

typedef struct Node *List;
struct Node{
    double data;
    List next;
};

void insertElement(List deck[], int index, double arr[], int i){
    List p,q;
    if (deck[index]->data > arr[i]){    //插入到第一个之前的情况单独分出来
        p = malloc(sizeof(struct Node));
        p->data = arr[i];
        p->next = deck[index];
        deck[index] = p;
    }else{
        p = deck[index]->next;
        q = deck[index];
        while(p != NULL && p->data <= arr[i]){ //这边这一个 <= 使之成为稳定排序
            q = p;
            p = p->next;
        }
        q->next = malloc(sizeof(struct Node));
        q->next->data = arr[i];
        q->next->next = p;      //插入到最后一个位置和插入到前面的位置合并，p为NULL时就是插入到最后一个位置
    }
}

void bucketSort(double arr[], int length){
    double max = -99999, min = 99999;
    for ( int i = 0; i < length; ++i) {
        if (arr[i] > max) max = arr[i];
        if (arr[i] < min) min = arr[i];
    }
    const int size = floor(max) - floor(min) + 1;
    List deck[size];            //建立结构体指针数组并初始化
    for (int i = 0; i < size; ++i) deck[i] = NULL;
    for (int i = 0; i < length; ++i) {
        int index = floor(arr[i]) - floor(min);
        if (deck[index] == NULL) {
            deck[index] = malloc(sizeof(struct Node));
            deck[index]->data = arr[i];
            deck[index]->next = NULL;
        }else{
            //原来不为空就内部排序
            insertElement(deck,index,arr,i);
        }
    }
    for (int i = 0, j = 0; i < size; ++i){
        List p = deck[i];
        while(p != NULL){
            arr[j] = p->data;
            j++;
            p = p->next;
        }
    }
}

//free我就不free了，写项目的时候要记得free

void printArray(double arr[], int length){
    for (int i = 0; i < length - 1; ++i) printf("%.1lf ", arr[i]);
    printf("%.1lf", arr[length - 1]);
}

int main(int argc, char const *argv[])
{
    double arr[MAX_SIZE];
    int tmp = 0;
    while(scanf("%lf", &arr[tmp]) != EOF && getchar() != '\n') tmp++;
    bucketSort(arr, tmp + 1);
    printArray(arr, tmp + 1);
    return 0;
}
