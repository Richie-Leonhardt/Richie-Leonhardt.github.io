BiTree CreatBiTree(BiTree T){
	char a[20];
	scanf("%s",a);
    if(strcmp(a,"#") == 0){
		T = NULL;
	}else{
		int input = 0;
		int i,j;
		for (i = 0; a[i] != '\0' && i < 20; ++i);				//这边两个for循环是将字符串的数字转换成int类型的
		for (i = i-1,j = 1; i >= 0; --i,j *= 10){
			input += ((int)a[i] - 48) * j;
		}
		T = (BiTree)malloc(sizeof(struct BiTNode));				//生成根结点
		T->data = input;
		T->lchild = CreatBiTree(T->lchild);                     //构造左子树
		T->rchild = CreatBiTree(T->rchild);                     //构造右子树
    }
    return T;
}

//非递归中序遍历
void InorderTraversal_NotRecursive(BiTree T){
	if ( T ){
		BiTree stack_arr[100];
		int top = -1;
		while( top != -1 || T != NULL ){
			while (T != NULL){
				stack_arr[++top] = T;
				T = T->lchild;
			}
			if (T == NULL){
				T = stack_arr[top--];
				printf(" %d",T->data );
				T = T->rchild;
			}
		}
	}
}

//后序遍历的非递归实现
void PostorderTraversal_NotRecursive(BiTree T){
	if ( T ){
		BiTree stack_arr[100];
		int top = -1;
		BiTree last;  	//记录最后一个输出值的节点
		while( top != -1 || T != NULL ){
			while (T != NULL){
				stack_arr[++top] = T;
				T = T->lchild;				//一直往左找，直到T为最左边的NULL
			}
			if (T == NULL){
				T = stack_arr[top];
				if (T->rchild == last || T->rchild == NULL){	//如果右子树刚刚输出过或是右子树为空则说明右边的子树输出完了，可以直接将这个根节点输出来了
					printf(" %d",T->data);
					last = T;
					top--;
					T = NULL; 
				}else{
					T = T->rchild;
				}
			}
		}
	}
}

//计算二叉树高度
int GetHeight( BiTree T ){
	if (T == NULL){
		return 0;
	}
	int a = GetHeight(T->lchild);
	int b = GetHeight(T->rchild);
	if (a > b){ 
		return a + 1;
	}
	return b + 1;
}

//计算二叉树叶子节点数目
int GetNumberOfLeaves( BiTree T ){
	if (T->lchild == NULL && T->rchild == NULL)
		return 1;
	return ((T->lchild == NULL) ? 0 : GetNumberOfLeaves(T->lchild)) + ((T->rchild == NULL) ? 0 : GetNumberOfLeaves(T->rchild));
}
